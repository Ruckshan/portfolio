and css

1.truncate {
2	/* Avoids text being rendered outside the container */
3	width: 100%;
4	overflow: hidden;
5	/* Avoid text going to multiple lines */
6	white-space: nowrap;
7	/* Sets the ... once the text overflows */
8	text-overflow: ellipsis;
9}
